def ats_scanner(resume, job_position):
    keywords = fetch_keywords(job_position)
    matches = count_matches(resume, keywords)
    structure_score = check_sections(['Experience', 'Skills', 'Education'])
    readability = calculate_flesch_score(resume)
    suggestions = generate_improvements(keywords - matches)
    score = (matches * 0.6) + (structure_score * 0.3) + (readability * 0.1)
    return score, suggestions

def job_matcher(resume, location, radius):
    resume_keywords = extract_keywords(resume)
    jobs = fetch_jobs(location, radius)
    for job in jobs:
        job['match_score'] = cosine_similarity(resume_keywords, job['description'])
    return sort_by_score(filter_remote(jobs))
